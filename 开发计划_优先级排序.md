class StatisticsManager:
    def __init__(self):
        self._weekly_trend_cache = None
        self._cache_timestamp = None

    def get_weekly_trend(self, days=7) -> List[Dict]:
        # 如果缓存有效(5分钟内),直接返回
        if (self._weekly_trend_cache and
            self._cache_timestamp and
            (time.time() - self._cache_timestamp) < 300):
            return self._weekly_trend_cache

        # 重新计算并缓存
        result = self._calculate_weekly_trend(days)
        self._weekly_trend_cache = result
        self._cache_timestamp = time.time()
        return result

```python
# gaiya/core/performance_monitor.py

class PerformanceMonitor:
    """性能监控器"""

    def __init__(self):
        self.start_time = None
        self.metrics = {}

    def record_startup_time(self):
        """记录启动时间"""
        startup_time = time.time() - self.start_time
        self.metrics['startup_time'] = startup_time

        if startup_time > 5.0:
            logger.warning(f"Startup time is slow: {startup_time:.2f}s")

    def record_memory_usage(self):
        """记录内存使用"""
        import psutil
        process = psutil.Process()
        memory_mb = process.memory_info().rss / 1024 / 1024
        self.metrics['memory_mb'] = memory_mb

        if memory_mb > 200:
            logger.warning(f"Memory usage is high: {memory_mb:.2f}MB")

    def record_ui_response_time(self, operation_name: str, duration: float):
        """记录UI操作响应时间"""
        if duration > 1.0:
            logger.warning(
                f"UI operation '{operation_name}' is slow: {duration:.2f}s"
            )
```
